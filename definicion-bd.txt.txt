tabla_persona:
  cedula_identidad      -> int (PK)
  nombres               -> varchar
  apellidos             -> varchar
  correo                -> varchar
  telefono              -> varchar (Mejor que int, para incluir c digos de  rea ej: +58)
  direccion_habitacion  -> text

tabla_ficha_medica_fisica: (Opcional, separada para no recargar datos b sicos)
  id_ficha              -> int (PK)
  cedula_persona_fk     -> int (FK hacia tabla_persona)
  tipo_sangre           -> enum
  peso                  -> int
  talla_camisa          -> varchar/set
  talla_pantalon        -> varchar/set
  talla_zapato          -> varchar/set
  alergias              -> json
  discapacidades        -> json
  afecciones_cronicas   -> json

tabla_estudiante:
  id_estudiante         -> int (PK)
  cedula_persona_fk     -> int (FK hacia tabla_persona)
  es_becado             -> tinyint/boolean

tabla_representante:
  id_representante      -> int (PK)
  cedula_persona_fk     -> int (FK hacia tabla_persona)

tabla_estudiante_representante: (Tabla intermedia por si un ni o tiene 2 representantes, o 1 representante tiene varios ni os)
  id_estudiante_fk      -> int
  id_representante_fk   -> int
  parentesco            -> varchar (ej. Madre, Padre, T o)

tabla_profesor:
  id_profesor           -> int (PK)
  cedula_persona_fk     -> int (FK hacia tabla_persona)

tabla_asignatura:
  id_asignatura     -> int (PK)
  id_fase_fk        -> int (FK hacia tabla_fase_academica. Ej: "Matem ticas" pertenece al "1er Semestre")
  nombre_asignatura -> varchar
  descripcion       -> text

tabla_profesor_asignatura: (Tabla intermedia M:N)
  id_profesor_fk    -> int
  id_asignatura_fk  -> int
  id_periodo_fk     -> int (FK hacia tabla_periodo_academico. Para saber EXACTAMENTE en qu  per odo ese profesor dio esa materia).
  
tabla_periodo_academico:
  id_periodo      -> int (PK)
  nombre_periodo  -> varchar (Ej: "A o Escolar 2024-2025", "Semestre 2024-I")
  fecha_inicio    -> date
  fecha_fin       -> date
  estado          -> varchar/enum (Ej: 'Planificaci n', 'Activo', 'Finalizado')
  
tabla_programa_educativo: (Representa la "Carrera" o el "Nivel General")
  id_programa     -> int (PK)
  nombre_programa -> varchar (Ej: "Ingenier a en Inform tica", "Educaci n Primaria")
  descripcion     -> text

tabla_fase_academica: (Representa los semestres, grados o trayectos)
  id_fase         -> int (PK)
  id_programa_fk  -> int (FK hacia tabla_programa_educativo)
  nombre_fase     -> varchar (Ej: "1er Semestre", "3er Grado", "Trayecto II")
  tipo_fase       -> varchar (Ej: "Semestre", "Grado", "Trayecto", "Trimestre")
  orden_secuencia -> int (Crucial: Un n mero ej. 1, 2, 3... para saber qu  fase va primero y poder calcular avances o promociones).
  
tabla_matricula: (La cabecera de la inscripci n)
  id_matricula          -> int (PK)
  id_estudiante_fk      -> int (FK hacia tabla_estudiante)
  id_periodo_fk         -> int (FK hacia tabla_periodo_academico)
  id_fase_fk            -> int (FK hacia tabla_fase_academica. Ej: Se inscribi  para el "2do Semestre")
  fecha_inscripcion     -> date/datetime
  estado_matricula      -> varchar/enum (Ej: 'Activa', 'Retirada', 'Suspendida', 'Completada')

tabla_matricula_detalle: (Las asignaturas espec ficas que va a ver)
  id_detalle            -> int (PK)
  id_matricula_fk       -> int (FK hacia tabla_matricula)
  id_profesor_asignatura_fk -> int (FK hacia la tabla intermedia que creamos antes. Esto es CLAVE: inscribe la asignatura conectada al profesor que la dicta en ese periodo)
  estado_asignatura     -> varchar/enum (Ej: 'Cursando', 'Aprobada', 'Reprobada', 'Retirada')
  nota_final_definitiva -> decimal (Un campo para guardar el promedio final una vez termine el periodo, por optimizaci n de consultas)
  
tabla_plan_evaluacion: (El profesor define C MO va a evaluar su clase)
  id_evaluacion         -> int (PK)
  id_profesor_asignatura_fk -> int (FK hacia la clase espec fica de ese periodo)
  nombre_evaluacion     -> varchar (Ej: "1er Parcial", "Proyecto Final", "Exposici n")
  descripcion_evaluacion -> varchar 
  porcentaje_peso       -> decimal (Ej: 20.00, 25.00. La suma total por clase deber a dar 100%)
  fecha_estimada        -> date (Cu ndo se realizar  la evaluaci n)

tabla_calificacion: (Las notas individuales de cada estudiante)
  id_calificacion       -> int (PK)
  id_detalle_matricula_fk -> int (FK hacia tabla_matricula_detalle. Identifica al estudiante en esa clase)
  id_evaluacion_fk      -> int (FK hacia tabla_plan_evaluacion. Identifica qu  examen es)
  nota_obtenida         -> decimal (Ej: 15.5, 18.0, o basado en la escala del pa s)
  observaciones         -> text (Por si el profesor quiere dejar un comentario, ej: "Entreg  tarde")

tabla_usuario: (Credenciales de acceso)
  id_usuario            -> int (PK)
  cedula_persona_fk     -> int (FK hacia tabla_datos_persona. Conecta la cuenta con la persona humana)
  nombre_usuario        -> varchar (Ej: 'jperez', o puede ser el mismo correo)
  contrasena_hash       -> varchar (NUNCA guardar contraseñas en texto plano, usar bcrypt o argon2)
  estado_cuenta         -> varchar/enum (Ej: 'Activo', 'Bloqueado', 'Pendiente_Activacion')
  ultimo_acceso         -> datetime (Para auditoría de seguridad)
